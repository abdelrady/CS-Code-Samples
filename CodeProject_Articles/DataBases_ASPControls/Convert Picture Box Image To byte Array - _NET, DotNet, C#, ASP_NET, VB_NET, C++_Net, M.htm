    private byte[] convertPicBoxImageToByte(System.Windows.Forms.PictureBox pbImage)

    {

    System.IO.MemoryStream ms = new System.IO.MemoryStream();

    pbImage.Image.Save(ms, System.Drawing.Imaging.ImageFormat.Jpeg);

    return ms.ToArray();

    }

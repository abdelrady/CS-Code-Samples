Introduction

MSDN describes a crossbar with following phrase : “ If the tuner card has more than one physical input, the Analog Video Crossbar filter enables the application to select which input is decoded and rendered.” (http://msdn2.microsoft.com/en-us/library/ms779290.aspx ) For example , you can select your video from SVideo/Composite or Tuner input via a crossbar in TV Cards. I have developed this sample using DXSDK9 under VS2003 using MFC as GUI. All Directshow interfaces I ve used are well-known interfaces exposed by Directshow developers except IAMCrossbar interface. It represents a crossbar device. I have used FlyVideo98 TV Card under Windows XP Pro SP2.

You can do this by building solution files in the path "YourInstallationPath:\DXSDK\Samples\C++\DirectShow\BaseClasses" After that step you must add include header files , libs and some source files to your solution. 5. Crossbar Functions

ArrangeCrossBarVideo() is the main function of this article. First of all you must get handle of IAMCrossbar filter by calling ICaptureGraphBuilder2::FindInterface . After finding requested pins , you simply call IAMCrossbar::Route method to route crossbar of capture device. 

BOOL CDSApplicationDlg::ArrangeCrossBarVideo ()
{
	HRESULT hr;
	int selectedIn = (1 + video_tuner_in->GetCurSel()) ;
	int selectedOut = (1 + video_tuner_out->GetCurSel()) ;
     	
	
	hr = CaptureGraph->FindInterface(&PIN_CATEGORY_CAPTURE,&MEDIATYPE_Interleaved, VideoSourceFilter ,IID_IAMCrossbar, (void **)&CrossbarFilter);
    
	if(hr != NOERROR)
	hr = CaptureGraph->FindInterface(&PIN_CATEGORY_CAPTURE, &MEDIATYPE_Video, VideoSourceFilter,IID_IAMCrossbar, (void **)&CrossbarFilter);

	LONG lInpin, lOutpin;
	hr = CrossbarFilter->get_PinCounts(&lOutpin , &lInpin); 

		BOOL IPin=TRUE; LONG pIndex=0 , pRIndex=0 , pType=0;
		
		while( pIndex < lInpin)
		{
			hr = CrossbarFilter->get_CrossbarPinInfo( IPin , pIndex , &pRIndex , &pType); 
		
				
	
			if( pType == selectedIn)
			//if( pType == PhysConn_Video_Composite)
					break;

			pIndex++;
		}
		
		BOOL OPin=FALSE; LONG pOIndex=0 , pORIndex=0 , pOType=0;
		
		while( pOIndex < lOutpin)
		{
			hr = this->CrossbarFilter->get_CrossbarPinInfo( OPin , pOIndex , &pORIndex , &pOType); 
			if( pOType == selectedOut)
			//if( pOType == PhysConn_Video_VideoDecoder)
				break;
		}

		this->CrossbarFilter->Route(pOIndex,pIndex); 

        this->CrossbarFilter->Release();
	return TRUE ;
}

6. Arranging Crossbar and Running Stream

Rest of project is typical Directshow graph-building code which calls crossbar functions to implement crossbar routing and ROT functions. 

void CDSApplicationDlg::OnBnClickedButtonRun()
{
	IVideoWindow  * VideoWindow;
	IMediaEventEx * MediaEvent ;
	IBaseFilter* VideoRenderer ;
	DWORD  GraphNumber = 4563;
	IMoniker * pMoniker;
        IRunningObjectTable *pROT;
        char wsz[128];
        HRESULT hr;
	OAFilterState fState ;
	IPin *pOut ;
	ICreateDevEnum *pSysDevEnum = NULL;
	int n = (sources->GetCurSel())+1 ;
	char* VideoSourceName = "VideoSource";
	int nCaptureDevice = 0;
	IEnumMoniker *pEnumCat = NULL;
	RECT rc;
	int VideoHeight = 150 ;

	CoCreateInstance (CLSID_FilterGraph, NULL, CLSCTX_INPROC,IID_IGraphBuilder, (void **) &(GraphBuilder));
    CoCreateInstance (CLSID_CaptureGraphBuilder2 , NULL, CLSCTX_INPROC,IID_ICaptureGraphBuilder2, (void **) &(CaptureGraph));
	GraphBuilder->QueryInterface(IID_IMediaControl,(LPVOID *) &(MediaControl));
	GraphBuilder->QueryInterface(IID_IVideoWindow, (LPVOID *) &(VideoWindow));
    GraphBuilder->QueryInterface(IID_IMediaEvent, (LPVOID *) &(MediaEvent));
	CaptureGraph->SetFiltergraph(GraphBuilder);
    if (!GraphBuilder )
        return ;
    if (FAILED(GetRunningObjectTable(0, &pROT)))
        return ;
	wsprintf(wsz, "FilterGraph %08x pid %08x\0", (DWORD_PTR)GraphBuilder, GetCurrentProcessId());
	   hr = CreateItemMoniker(L"!", ConvertAnsiCharToUnicodeChar(wsz), &pMoniker);
  	if (SUCCEEDED(hr)) 
    {
       
        hr = pROT->Register(ROTFLAGS_REGISTRATIONKEEPSALIVE, GraphBuilder, pMoniker,&( GraphNumber));
        pMoniker->Release();
    }

    pROT->Release();
	
hr = CoCreateInstance(CLSID_SystemDeviceEnum, NULL, CLSCTX_INPROC_SERVER,
    IID_ICreateDevEnum, (void **)&pSysDevEnum);
hr = pSysDevEnum->CreateClassEnumerator(CLSID_VideoInputDeviceCategory, &pEnumCat, 0);
if (hr == S_OK) 
{
    IMoniker *pMoniker = NULL;
    ULONG cFetched;
    while(pEnumCat->Next(1, &pMoniker, &cFetched) == S_OK)
    {
        nCaptureDevice++ ;
		
	IPropertyBag *pPropBag;
        
		hr = pMoniker->BindToStorage(0, 0, IID_IPropertyBag, 
            (void **)&pPropBag);
        if (SUCCEEDED(hr))
        {
           
            VARIANT varName;
            VariantInit(&varName);
            hr = pPropBag->Read(L"FriendlyName", &varName, 0);

            if (SUCCEEDED(hr))
            {
  
		if(nCaptureDevice == n)
		{
		pMoniker->BindToObject(0,0,IID_IBaseFilter, (void**)&(VideoSourceFilter));
		GraphBuilder->AddFilter(VideoSourceFilter,ConvertAnsiCharToUnicodeChar(VideoSourceName));
		}
		}
            VariantClear(&varName);

            IBaseFilter *pFilter;
            hr = pMoniker->BindToObject(NULL, NULL, IID_IBaseFilter,
                (void**)&pFilter);
          
            pPropBag->Release();
        }
        pMoniker->Release();
    }
    pEnumCat->Release();
}
pSysDevEnum->Release();

	hr = CoCreateInstance(CLSID_VideoRenderer, 0, CLSCTX_INPROC_SERVER,IID_IBaseFilter, reinterpret_cast(&VideoRenderer));
    if (SUCCEEDED(hr))
    {
		hr = GraphBuilder->AddFilter(VideoRenderer, L"Video Renderer");
          
    }

	ConnectFilters(VideoSourceFilter,VideoRenderer,1,1);
	
			ArrangeCrossBarVideo();
			ArrangeCrossBarAudio();

	VideoWindow->put_Owner((OAHWND)this->m_hWnd);
  	VideoWindow->put_WindowStyle(WS_CHILD| WS_CLIPSIBLINGS | WS_CLIPCHILDREN |WS_OVERLAPPED);
	VideoWindow->put_MessageDrain((OAHWND)this->m_hWnd);
    VideoWindow->put_Visible(OATRUE);
	GetClientRect( &rc);
	 VideoWindow->SetWindowPosition(0, VideoHeight, rc.right, rc.bottom-VideoHeight);

	pOut = GetOutPin(VideoRenderer,1);
	GraphBuilder->Render(pOut);

	

	VideoSourceFilter->Release();  
	MediaControl->GetState(NULL,&fState);

	if( fState != State_Running )
	{
			MediaControl->Run();
	}

	run_button->EnableWindow(FALSE);
	 bInit =TRUE ;

	return ;

///////////////////////////PARAMETRIC FUNCTION///////////////////////
void Parametric(HDC hdc, int a,int b,int h,int k,COLORREF col)
{
	double theta=0;
	int x,y;
	double theta_end=44/7;
	while(theta<theta_end)
	{
		x=(int)(a*cos(theta));
		y=(int)(b*sin(theta));
		SetPixel(hdc,x+h,y+k,col);
		SetPixel(hdc,-x+h,-y+k,col);
		SetPixel(hdc,-x+h,y+k,col);
		SetPixel(hdc,x+h,-y+k,col);
		theta+=0.05;
	}
}

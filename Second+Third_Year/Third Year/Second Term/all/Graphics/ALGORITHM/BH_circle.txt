void BH_CIRCLE(HDC hdc,int cx,int cy,int r,COLORREF col)
{
	int x=0,y=r,d=3-2*r;
	while(x<=y)
	{
		SetPixel(hdc,x+cx,y+cy,col);
		SetPixel(hdc,y+cy,x+cx,col);
		SetPixel(hdc,-y+cy,x+cx,col);
		SetPixel(hdc,-x+cx,y+cy,col);
		SetPixel(hdc,-x+cx,-y+cy,col);
		SetPixel(hdc,x+cx,-y+cy,col);
		SetPixel(hdc,y+cy,-x+cx,col);
		SetPixel(hdc,-y+cy,-x+cx,col);	
		if(d<0)
			d=d+4*x+6;
		else
		{
			d=d+4*(x-y)+10;
			y--;
		}
		x++;
	}
}

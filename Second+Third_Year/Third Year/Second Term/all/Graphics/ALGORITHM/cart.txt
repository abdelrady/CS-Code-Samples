void Cart(HDC hdc, int cx,int cy, int r,COLORREF col)
{
	int x=0;
	int xend=(r/sqrt(2));
	int y;
	while(x<xend)
	{
		y=sqrt((r*r) - (x*x));
		SetPixel(hdc,x+cx,y+cy,col);
		SetPixel(hdc,y+cy,x+cx,col);
		SetPixel(hdc,-y+cy,x+cx,col);
		SetPixel(hdc,-x+cx,y+cy,col);
		SetPixel(hdc,-x+cx,-y+cy,col);
		SetPixel(hdc,x+cx,-y+cy,col);
		SetPixel(hdc,y+cy,-x+cx,col);
		SetPixel(hdc,-y+cy,-x+cx,col);
		x++;
	}
}

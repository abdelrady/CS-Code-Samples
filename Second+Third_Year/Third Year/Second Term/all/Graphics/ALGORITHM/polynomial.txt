void Trig(HDC hdc,int cx,int cy,int r,COLORREF col)
{
	double Theta=0;
	while(Theta<=22.0/28)
	{
		int x=(int)(r*cos(Theta));
		int y=(int)(r*sin(Theta));
		SetPixel(hdc,x+cx,y+cy,col);
		SetPixel(hdc,y+cy,x+cx,col);
		SetPixel(hdc,-y+cy,x+cx,col);
		SetPixel(hdc,-x+cx,y+cy,col);
		SetPixel(hdc,-x+cx,-y+cy,col);
		SetPixel(hdc,x+cx,-y+cy,col);
		SetPixel(hdc,y+cy,-x+cx,col);
		SetPixel(hdc,-y+cy,-x+cx,col);
        Theta=Theta+(7.0/(2*22.0*r));
	}
}
